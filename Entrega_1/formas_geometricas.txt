	variáveis:
	
	    ponto1;
	    ponto2;
	    coeficiente_angular;
	    cumprimento;
	    
	    
	funções membro:
	
	    criar_linha(x1, y1, x2, y2){
	        ponto1 = ponto.criar_ponto(x1, y1);
	        ponto2 = ponto.criar_ponto(x2, y2);
	        
	        //se x1 e x2 forem iguais a reta e vertical
	        if(x1 != x2){
	            coeficiente_angular = (y1 - y2) / (x1 - x2);
	        }
	        
	        else{
	            coeficienten_angular = "indefinido";
	        }
	       
	    }
	    
	    cumprimento_linha(linha){
	        cumprimento = sqrt((linha.ponto1.y - linha.ponto2.y)** + (linha.ponto1.x - linha.ponto2.x)**2)
	    }
